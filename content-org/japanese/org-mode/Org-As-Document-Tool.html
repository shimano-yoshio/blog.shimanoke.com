2 Org-mode の中で図を書く

図を描くのではなく、図をテキストで書きます。

2.1 blockdig を使う

nwdiag {
    network FJWAN {
        gateway [address = "10.37.138.1"];
    }
    network Subnet-5F {
        address = "10.37.138.0/24"

        gateway [address = ".1"];
        wathcer [address = ".120"];
        gitlab [address = ".82"];
        ansible [address = ".56"];
        tomcat1 [address = ".60"];
        }
    network natnetwork {
        address = "10.0.2.0/24"

        wathcer [address = ".101"];
        gitlab [address = ".102"];
        ansible [address = ".103"];
        tomcat1 [address = ".104"];
    }

}
